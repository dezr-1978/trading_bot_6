# utils/time.py

import time
from datetime import datetime, timedelta
from loguru import logger


TIMEFRAME_SECONDS = {
    "1m": 60,
    "3m": 180,
    "5m": 300,
    "15m": 900,
    "30m": 1800,
    "1h": 3600,
    "4h": 14400,
    "1d": 86400,
}


def sleep_seconds(seconds: int):
    """
    Простий sleep з логом
    """
    logger.debug(f"[TIME] Sleeping for {seconds} seconds")
    time.sleep(seconds)


def sleep_to_next_candle(timeframe: str):
    """
    Засипає до початку наступної свічки
    """
    if timeframe not in TIMEFRAME_SECONDS:
        raise ValueError(f"Unsupported timeframe: {timeframe}")

    tf_sec = TIMEFRAME_SECONDS[timeframe]
    now = datetime.utcnow()

    # секунд з початку епохи
    now_ts = int(now.timestamp())
    next_candle_ts = ((now_ts // tf_sec) + 1) * tf_sec

    sleep_for = next_candle_ts - now_ts

    if sleep_for > 0:
        logger.info(
            f"[TIME] Sleeping until next {timeframe} candle "
            f"({sleep_for}s)"
        )
        time.sleep(sleep_for)


def is_new_candle(last_candle_time: datetime, timeframe: str) -> bool:
    """
    Перевіряє, чи почалась нова свічка
    """
    if timeframe not in TIMEFRAME_SECONDS:
        raise ValueError(f"Unsupported timeframe: {timeframe}")

    tf_sec = TIMEFRAME_SECONDS[timeframe]
    return datetime.utcnow() >= last_candle_time + timedelta(seconds=tf_sec)
